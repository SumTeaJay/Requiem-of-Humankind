on_actions = {
	on_startup = {
		effect = {
			set_variable = { intention_point = 0}
		}

	}

	on_daily_ODS = {
		effect = {
			if = {
				limit = { has_country_flag = third_phase}
				clamp_variable = {
					var = intention_point
					min = 0
					max = 100
				}
			}	
			else = {
				clamp_variable = {
					var = intention_point
					min = 0
					max = 40
				}
			}	
		}
	}

	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = ODS }
					FROM.FROM = { 
						NOT = { 
							is_core_of = ODS 
							has_state_flag = second_level_of_authority
							has_state_flag = first_level_of_authority
						} 
					}
				}
				FROM.FROM = {
					add_dynamic_modifier = { modifier = ODS_new_land }
				}
			}
			if = {
				limit = {
					FROM.FROM = { 
						NOT = { 
							is_core_of = ODS 
							is_controlled_by = ODS
						} 
						has_dynamic_modifier = { modifier = ODS_new_land }
					}
				}
				FROM.FROM = {
					remove_dynamic_modifier = { modifier = ODS_new_land }
				}
			}
		}
	}
}
