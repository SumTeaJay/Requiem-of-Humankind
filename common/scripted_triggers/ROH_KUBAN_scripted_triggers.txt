Kuban_war_has_not_begun = {
	AND = {
		STW = {
			NOT = { has_war_with = KUH }
			NOT = { has_war_with = FSR }
		}
		KUH = {
			NOT = { has_war_with = STW }
			NOT = { has_war_with = FSR }
		}
		FSR = {
			NOT = { has_war_with = STW }
			NOT = { has_war_with = KUH }
		}
		date > 2022.06.28 
	}
}
